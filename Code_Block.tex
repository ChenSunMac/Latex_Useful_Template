 $\bullet$ Example:  Booth Algorithm in Verilog:

\begin{lstlisting}
module booth_multi (A, B, P, CLK, reset)

input [7:0] A, B;
input CLK, reset;
output reg [15:0] P;
reg [7:0] N;
reg [15:0] N_prime; 
reg [8:0] A_prime;
integer i;

always @ (posedge CLK)
begin
if (~reset) beign
P <= 16'b0;
A_prime <= {A, 1'b0};
end
else begin
for (i = 0; i < 8; i = i + 1)
begin
if (A_prime[1:0] == 2'b 01)
N <= B;
if (A_prime[1:0] == 2'b 10)
N <= -B;
N_prime <= {N, 8'b 0};
P  <= P + N_prime;
A_prime <= 	A_prime >>> 1;
P <= P >>> 1;
end
end		
end

\end{lstlisting}
